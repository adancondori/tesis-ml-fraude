\section{Documentación del Dataset}

\subsection{Descripción de Variables}
\begin{itemize}
    \item \textbf{transaction\_id}: Identificador único de transacción
    \item \textbf{monto}: Valor de la transacción en USD
    \item \textbf{canal}: Canal de pago (web, móvil, POS)
    \item \textbf{gateway}: Pasarela de pago utilizada
    \item \textbf{pais}: País de origen de la transacción
    \item \textbf{fraude}: Variable objetivo (0=legítimo, 1=fraude)
\end{itemize}
